    消除类游戏

    不如取名叫折磨类游戏好了。我最开始写的时候只申请了一个一位数组(真是个傻逼)，想法是在上下左右四周设置墙，后来发现只需要右下设墙就行了，而且因为这么一搞我倒还为怎么去初始化大伤脑筋，吐了。

    在检测是否可以消除时用了一个递归的方法，事实上检测时只需要定一个点向右和向下搜索就好了，甚至在与右边界和下边界的距离小于2时便可以停止搜索，确实代码也不多，因为向右和向下的代码基本相同，实际上定义一个方向就好了。

                #define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
#include <assert.h>

#define RIGHT(i, p) p[i] == -1 ? i : i + 1
#define DOWN(i, p) p[i] == -1 ? i : i + column
#define R_D(i, p, state) state ? RIGHT(i, p) : DOWN(i, p) 
int* create();
void output(int*);
void traverse();
void mark(int, int, int);
void merge();

int row, column;
int* map = NULL;
int* tmp = NULL;
void main() {
	scanf("%d %d", &row, &column);
	map = create();
	traverse();
	merge();
	output(map);
	
	free(map);
	free(tmp);
}

void traverse() {
	int i = column + 1;
	while (map[i] != -1) {
		while (map[i] != -1) {
			mark(i, map[i], 1);//1表示向右搜索
			mark(i, map[i], 0);//0表示向下搜索
			i = RIGHT(i, map);
		}
		i += 2;
	}
}

void mark(int pos, int number, int state) {
	static int flag;
	static int count;
	flag++;
	if (map[pos] == number) {
		count++;
		mark(R_D(pos, map, state), number, state);
	}
	tmp[pos] = count > 2 && map[pos] == number ? 1 : tmp[pos];
	count = !(--flag) ? 0 : count;
}

void merge() {
	int i;
	for (i = 0; i < row * column; i++)
		map[i] = tmp[i] ? 0 : map[i];
}

    N阶顺转方阵

    也是采用图的遍历的方法(好像)，代码几乎大同小异，不过这个需要规定四个方向，按照右下左上的顺序一步一步走，如果走到边界或者已走过的地方就换下一个方向尝试，我的判断终止的方法就是规定循环的次数，简单粗暴。

    			#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
#define RIGHT(pos) map[pos] != -1 ? pos + 1 : pos
#define LEFT(pos) map[pos] != -1 ? pos - 1 : pos
#define DOWN(pos) map[pos] != -1 ? pos + length : pos
#define UP(pos) map[pos] != -1 ? pos - length : pos

void iter();
void create();
void output();

int* map = NULL;
int length;
enum direction { right, down, left, up };
int pos;
int ij;
enum direction state;

int main() {
	create();
	output();
	free(map);
	return 0;
}

void create() {
	int i;
	scanf("%d", &length);//创建一个（n+2）^2的方阵，边界设-1。
	length += 2;
	map = (int*)calloc(length * length, sizeof(int));
	for (i = 0; i < length; i++)
		map[i] = -1;
	for (i = length - 1; i < length * length; i += length)
		map[i] = -1;
	for (i = 0; i <= length * (length - 1); i += length)
		map[i] = -1;
	for (i = length * (length - 1); i < length * length; i++)
		map[i] = -1;
	pos = length + 1;
	ij = 1;
	state = right;
	for (i = 0; i < (length - 2) * (length - 2); i++)
		iter();
}

void output() {
	int i = length + 1;
	while (map[i] != -1) {
		while (map[i] != -1) {
			printf("%d", map[i++]);
			if (map[i] != -1)
				putchar(' ');
		}
		putchar('\n');
		i += 2;
	}
}

void iter() {
	map[pos] = ij++;
	switch (state) {
	case right:
		pos = RIGHT(pos);
		if (map[RIGHT(pos)])
			state = down;
		break;
	case down:
		pos = DOWN(pos);
		if (map[DOWN(pos)])
			state = left;
		break;
	case left:
		pos = LEFT(pos);
		if (map[LEFT(pos)])
			state = up;
		break;
	case up:
		pos = UP(pos);
		if (map[UP(pos)])
			state = right;
	}
}

void fun(char* str) {
	char* p = str;
	int num = 0, i = 0, tmp = 0;
	while (*p != '\0') {
		if (*p <= '9' && *p >= '0') {
			while (*p != ' ' && *p != '\0') {
				num = num * 10 + *p - '0';
				p++;
			}
			stack_push(num);
			num = 0;
		}
		else {
			switch (*p) {
			case '+':
				stack_push(stack_pop() + stack_pop());
				break;
			case '-':
            if (p[1] <= '9' && p[1] >= '0') {
					p++;
					while (*p != ' ' && *p != '\0') {
						num = num * 10 + *p - '0';
						p++;
					}
					stack_push(-num);
					num = 0;
					p++;
					continue;
				}
				tmp = stack_pop();
				stack_push(stack_pop() - tmp);
				break;
			case '*':
				stack_push(stack_pop() * stack_pop());
				break;
			case '/':
				tmp = stack_pop();
				stack_push(stack_pop() / tmp);
				break;
			}
			p++;
		}
		p++;
	}
	printf("%d", stack_pop());
	return;
}

void stack_init() {
	stack.size = 0;
	stack.top = NULL;
	return;
}

void stack_push(int num) {
	Stacknode* tmp;
	tmp = (Stacknode*)calloc(1, sizeof(Stacknode));
	if (stack.size == 0)
		tmp->pre = NULL;
	else 
		tmp->pre = stack.top;
	tmp->next = NULL;
	tmp->data = num;
	stack.top = tmp;
	stack.size++;
	return;
}

int stack_pop() {
    if(stack.size == 0){
        printf("error!");
        exit(0);
    }
	int num;
	Stacknode* tmp;
	num = stack.top->data;
	tmp = stack.top;
	stack.top = stack.top->pre;
	stack.size--;
	free(tmp);
	return num;
}

void stack_destroy() {
	while (stack.size)
		stack_pop();
}
